@description
map = new ImageMap('map', 'Base Layer', 'img/garnet1.jpg', 500);

// map = new ImageMap({div:'map', name:'Base Layer', url: 'img/garnet1.jpg', scale: 500});
var feature = ImageMap.MakeFeature('Little Feature', 'img/garnet2.jpg',10,10,200,300);
var layer = ImageMap.MakeLayer("Little Feature Holder");
map.addFeatureToLayer(feature,layer);
map.addLayer(layer);
var layer2 = ImageMap.MakeLayer("Other Layer");
var feature2 = ImageMap.MakeFeature('Little Feature Recursive', 'img/garnet1.jpg',-10,-10,100,100);
var feature3 = ImageMap.MakeFeature('Little Feature 3', 'img/widelong.jpg',200,200,50,50);
var feature4 = ImageMap.MakeFeature('Jerry', 'img/widelong.jpg',200,200,50,50);
map.addFeatureToLayer([feature2,feature3], layer2);
map.addLayer(layer2);
layer2.lockLayer();
layer2.setIndex(1)
layer.setIndex(2);
layer.addMetaData("This is a layer that holds things")
map.moveFeature(feature, 40,-10);
map.removeFeature(feature3); // Invalidates feature
map.addFeatureToLayer(feature4, layer2)
feature4.addMetaData("Uncle Jerry");
map.save();
