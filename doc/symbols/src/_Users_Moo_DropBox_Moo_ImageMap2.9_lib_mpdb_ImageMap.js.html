180 			 	var fe = this.features[f];
181 			 	if (!fe.attributes.graphicHeight) continue;
182 			 	
183 			 				 	var s = fe.attributes.graphicHeight / this.features[f].attributes.graphicWidth;
184 			 	var scalingFactor = scale / this.width;;
185 			 	fe.attributes.graphicWidth *= scalingFactor;
186 			 	fe.attributes.graphicHeight *=  scalingFactor;
187 			 	var c = fe.geometry.bounds.clone();
188 			 	fe.geometry.move(0,0);
189 			 	fe.geometry.move(c.left * scalingFactor - c.left ,c.top * scalingFactor - c.top);

198 		 	}

201 			 	this.width = scale;
202 
203 		 	for (var l in this.layers){

205 		 	}
